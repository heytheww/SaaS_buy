# 关于redis封装 rdb.go 的使用
```
package main

import (
	"context"
	"fmt"
	"go_web/mydb"
	"log"
	"os"
	"path/filepath"
	"time"
)

func main() {
	rdb := mydb.RDB{Addr: "localhost:6379", DBIndex: 0, Password: ""}
	err := rdb.InitRDB()
	if err != nil {
		log.Fatal(err)
	}
	ctx := context.Background()

	rdb.Set(ctx, "ww", 12, 300*time.Second)
	s, err2 := rdb.Get(ctx, "a")
	if err2 != nil {
		log.Fatal(err2.Error())
	}
	fmt.Println(s)

	mq := rdb.InitMQ("mq", 1000)
	rdb.AddMsg(ctx, &mq, "product_id", "1001")
	rdb.AddMsg(ctx, &mq, "product_id", "1002")
	rdb.AddMsg(ctx, &mq, "product_id", "1003")
	rdb.AddMsg(ctx, &mq, "product_id", "1004")
	rdb.AddMsg(ctx, &mq, "product_id", "1005")

	rdb.GetGroup(ctx, &mq, "cg1")
	xs, err2 := rdb.GetMsgByGroup(ctx, &mq, "c1")
	if err2 != nil {
		log.Fatal(err2)
	}
	for _, v := range *xs {
		fmt.Println(v.Stream)
		for _, v2 := range v.Messages {
			fmt.Println(v2.ID)
			fmt.Println(v2.Values["product_id"])
		}
	}

	pwd, _ := os.Getwd() // 获取当前所在工作目录
	f_path := filepath.Join(pwd, "mydb", "stock.lua")
	buf, err3 := os.ReadFile(f_path)
	if err3 != nil {
		log.Fatalln(err3)
	}
	keys := []string{"stock", "1001"}
	values := []interface{}{}
	num, err4 := rdb.RunLua(ctx, string(buf), keys, values)
	if err4 != nil {
		log.Fatalln(err4)
	}
	switch num {
	case -1:
		fmt.Println("库存不限")
	case -2:
		fmt.Println("商品不存在")
	case 0:
		fmt.Println("库存不足")
	default:
		fmt.Println("扣前库存还剩：", num)
	}
}
``` 
